# configuration example
# Copy this content to a file named .env.local in the root directory

# Google Gemini API Key (Server-side)
# Get one here: https://aistudio.google.com/app/apikey
GOOGLE_API_KEY=your_gemini_api_key_here

# Mapbox Public Token (Client-side)
# MUST start with NEXT_PUBLIC_ to be visible to the browser
# Get one here: https://mapbox.com/
NEXT_PUBLIC_MAPBOX_TOKEN=your_mapbox_token_here
